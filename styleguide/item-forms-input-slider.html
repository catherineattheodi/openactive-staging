4.1.9 Slider

Example

50 
<form oninput="x.value=parseInt(a.value)">
  <label for="fader">Volume</label>
  <input type="range"
    name="a"
    id="fader"
    min="1"
    max="100"
    value="50"
    step="1"
    aria-valuemin="1"
    aria-valuemax="100"
    aria-valuenow="50"
    oninput="outputUpdate(value)">
  <output name="x" for="fader" id="volume">50</output>
</form>
